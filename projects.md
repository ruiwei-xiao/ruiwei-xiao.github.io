  <!-- ===== Project 2: Pedagogical Prompting ===== -->
  <div class="project-row">
    <!-- 左侧：大块 Project 区域 -->
    <div class="project-hero">
      <div class="project-hero-inner">
        <div class="project-pill">Project</div>
        <h3>Pedagogical Prompting</h3>
        <p>
          Studying how learners and teachers prompt AI tools, and how to design
          scaffolds that turn prompting into a learnable skill.
        </p>
        <ul>
          <li>Contrastive examples and feedback for prompts.</li>
          <li>Multi-modal prompting in programming and AI literacy tasks.</li>
          <li>Design implications for classroom use of GenAI.</li>
        </ul>
      </div>
    </div>
